答案:

#include <stdio.h>
#include <stdlib.h>
#define N 80
int CountWords(char str[]);
int main()
{
    char str[N+1];
    printf("Input a string:");
    gets(str);
    printf("Numbers of words = %d\n",CountWords(str));
    return 0;
}
int CountWords(char str[])
{
    int i,j=0,k=0,count=0;
    for(i=0;str[i]==' ';i++);
    j=i-1;
    for(i=0;str[i]!='\0';i++)
    {
        if(str[i]==' ')
        {
            for(k=i;str[k]==' ';k++);
            i=k-1;
            //printf("%c\n",str[i]);
            if(i>j)
            {
                count++;
                j=i;
            }
        }
    }
    if(i>j&&str[i-1]!=' ')
    {
        count++;
        j=i;
    }
    return count;
}
