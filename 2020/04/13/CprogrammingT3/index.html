C语言编程T3(妙)

思路:

思路一:暴力破解(正向思维)

不断尝试各种数字,逐一尝试
但是由于n不确定,所以不知道要嵌套多少for和准备多少临时变量
故不可使用穷举法

思路二:回溯(逆向思维)

从后往前观察思考，找到其规律
采用数组,从后往前,逐一尝试,不符或者满足则采用不同的回溯，直至得出所有解

题目:

答案:

核心:
找规律 : 正向、逆向、解中有规律

#include <stdio.h>
#include <stdlib.h>
#define N 10

int main()
{
    int i[N];
    int sum,n,total,k,flag=0,count=0;
    printf("Please enter requried terms (<=10):");
    scanf("%d",&n);
    printf( "                             their sum:");
    scanf("%d",&total);
    sum=0;
    k=n;
    //从后往前推导
    i[n]=1;
    printf("There are following possible series:\n");
    while(1)
    {

        if(sum+i[k]<total)
        {
            if(k<=1)
            {
                i[1]=total-sum;
                flag=1;
            }
            else
            {
                sum+=i[k];
                k--;
                //从最小值开始往前赋值
                i[k]=i[k+1];
                continue;
                //继续while循环判断
            }
        }
        //sum在中间已经大于可total
        else if(sum+i[k]>total||k!=1)
        {
            //想办法回溯
            //减去上一个sum加的i[k]
            sum-=i[++k];
            flag=0;
        }
        else//sum+i[k]==total&&k==1
        {
            flag=1;
        }
        if(flag)
        {
             printf("[%d]:",++count);
            for(flag=1;flag<=n;++flag)
                printf("%d",i[flag]);
            printf("\n");
        }
        //判断i[n]是否已经变化到极致
        if(++k>n)
        {
            break;
        }
        //往后回溯
        sum-=i[k];
        //不断试探
        i[k]++;
    }
    return 0;
}
