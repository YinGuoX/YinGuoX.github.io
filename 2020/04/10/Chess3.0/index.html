进一步对跳跳马2.0程序输入和输出界面的优化:

2.进行了代码局部优化

1.代码实现:

#include <stdio.h>
#include <stdlib.h>
#define N 64
int IsSamePoint(int x,int b[],int n);
int main()
{
    int a[N][N];
    int b[N];//用来存储已经经过的点
    int c[N];//记录上一个顶点的列号
    int i,j,count=1,row,col,n;
    printf("输入顶点数目:\n");
    scanf("%d",&n);
    printf("n=%d\n",n);
    for(i=0;i<n;i++)
    {
        b[i]=-1;
        c[i]=-1;
    }
    b[0]=0;
    c[0]=0;
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            a[i][j]=0;
        }
    }
    do
    {
        printf("请输入对应的边(x y 1)(输入一半即可,-1表示结束):\n");
        scanf("%d%d",&i,&j);
        scanf("%d",&a[i][j]);
        //printf("(%d,%d)=%d\n",i,j,a[i][j]);
        //printf("res=%d\n",res);
        a[j][i]=a[i][j];
    }while(i!=-1);
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            printf("%4d",a[i][j]);
        }
        printf("\n");
    }
    row=0;
    col=1;
    while(count<n)
    {
        if(a[row][col]!=0&&IsSamePoint(col,b,n))
        {
            printf("(%d,%d)-->(%d,0)\n",row,col,col);
            b[count]=col;
            count++;
            row=col;
            col=0;
        }else
        {
            if(col<n-1)
            {
                col++;
            }else
            {
                //回溯
                do{
                    row=b[count-2];
                    col=b[count-1]+1;
                    printf("col=%d\n",col);
                    b[count-1]=-1;
                    count--;
                }while(col>=n);
                printf("回溯到:(%d,%d)\n",row,col-1);
                //printf("越界了\n");
            }
        }

    }
    for(i=0;i<n;i++)
    {
        printf("%d  ",b[i]);
    }
    return 0;
}
//在b中判断是否有与x相同的元素
//有相同就返回0
int IsSamePoint(int x,int b[],int n)
{
    int i,flag=1;
    for(i=0;i<n;i++)
    {
        //printf("b=%d,x=%d\n",b[i],x);
        if(b[i]==x)
        {
            flag=0;
            return flag;
        }
    }
    return flag;
}
