C语言改错T3

易错点:

分清:

1.指针数组: int *p[4]
常用于字符串交换,因为相比strcpy()更加节省时间和空间
故使用指针数组进行交换效率更高索引技术、索引排序

函数中使用:
void Fun(int *p[M]);
Fun(p);

2.二维数组的 行指针 : int (*p)[4]

p[i]==*(p+i):也就是p[i]的地址！！！
p[i][j] == *(p[i]+j) == *(*(p+i)+j) == (*(a+i))[j]
&p[i][j] == p[i]+j == *(p+i)+j
函数中使用:
void Fun(int (*p)[M]);
Fun(p);//p == p[0] == p[0]的地址

3.二维数组的列指针 : int *p; p=*a

p=*a <==> p=a[0] <==> p=&a[0][0]
a[i][j]=*(p+i*n+j)
&a[i][j]=p+i*n+j
函数中使用:
void Fun(int *p);
Fun(*p);

题目:

答案:

核心:指针数组的妙用！！！

#include<stdio.h>
#include<string.h>
void Input(char *p[], int n);
void Sort(char *p[], int n);
void Print(char *p[], int n);

int main()
{                
    char str[20][40];
    char *pstr[20];
    int i, n;
    printf("Input n(n<=20):\n");
    scanf("%d", &n);
    //注意要清空缓冲区里的空格
    getchar();
    for (i = 0; i < n; i++)
    {          
        //指针数组要初始化
        pstr[i] = str[i];           //2
    }
    Input(pstr, n);
    Sort(pstr, n);
    printf("Results:\n");
    Print(pstr, n);
    return 0;
}                
void Input(char *p[], int n)
{                
    int i;

    for (i = 0; i < n; i++)
    {                
        gets(p[i]);
    }
}                
void Sort(char *p[], int n)
{                

    char *t;                              //1
    int i, j;

    for (i = 0; i < n - 1; i++)
    {                
        //注:j=i+1而不是1
        for (j = i + 1; j < n; j++)
        {                
            if (strcmp(p[j], p[i]) < 0)    //2
            {       
                //直接用指针数组来简化strcpy()
                t = p[i];
                p[i] = p[j];
                p[j] = t;
            }
        }
    }
}                
void Print(char *p[], int n)
{                
    int i;

    for (i = 0; i < n; i++)
    {                
        printf("%s\n", p[i]);
    }
}
